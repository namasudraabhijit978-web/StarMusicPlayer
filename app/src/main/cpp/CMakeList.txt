cmake_minimum_required(VERSION 3.22.1)
project("star_audio_engine" CXX)

# Upgraded to C++20
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Optimization & Compiler Flags (Disabled exceptions & rtti for raw DSP speed)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -O3 -ffast-math -Wall -fno-exceptions -fno-rtti")

# Requires 'buildFeatures { prefab = true }' in app/build.gradle.kts
find_package(oboe REQUIRED CONFIG)

# Scalable Source Globbing (Auto-detects all C++ files in folders)
file(GLOB_RECURSE NATIVE_SRC
    "dsp/*.cpp"
    "engine/*.cpp"
    "jni/*.cpp"
    "*.cpp"
)

add_library(star_audio_engine SHARED ${NATIVE_SRC})

# Removed OpenSLES. Oboe handles internal routing.
target_link_libraries(star_audio_engine
    oboe::oboe
    log
    android
)
